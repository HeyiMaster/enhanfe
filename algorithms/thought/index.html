    function flip(root) {
      const rec = node => {
        if (!node) return;
        const left = rec(node.left);
        const right = rec(node.right);
        const res = {
          value: node.value,
        };
        if (right) res.left = right;
        if (left) res.right = left;
        return res;
      };
      return rec(root);
